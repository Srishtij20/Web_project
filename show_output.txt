<script>
  const locationName = "<%= listing ? listing.location + ', ' + listing.country : '' %>";
  if (locationName) {
    fetch(`https://nominatim.openstreetmap.org/search?format=json&q=${encodeURIComponent(locationName)}`)
      .then(r => r.json())
      .then(data => {
        if (data.length > 0) {
          const map = L.map("map").setView([data[0].lat, data[0].lon], 12);
          L.tileLayer("https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png", {
            attribution: "© OpenStreetMap contributors"
          }).addTo(map);
          L.marker([data[0].lat, data[0].lon]).addTo(map)
            .bindPopup("<b><%= listing ? listing.title : '' %></b>").openPopup();
        }
      });
  }

  function shareListing() {
    navigator.clipboard.writeText(window.location.href);
    alert("Link copied to clipboard!");
  }

  const ta = document.getElementById("reviewComment");
  if (ta) ta.addEventListener("input", () => {
    document.getElementById("charCount").textContent = ta.value.length;
  });

  const pricePerNight = <%= listing ? listing.price : 0 %>;
  function calcPrice() {
    const ci = document.getElementById("checkIn").value;
    const co = document.getElementById("checkOut").value;
    if (!ci || !co) return;
    const nights = Math.round((new Date(co) - new Date(ci)) / 86400000);
    if (nights <= 0) return;
    const subtotal = pricePerNight * nights;
    const fee = Math.round(subtotal * 0.05);
    const gst = Math.round((subtotal + fee) * 0.18);
    const total = subtotal + fee + gst;
    document.getElementById("nights").textContent = nights;
    document.getElementById("subtotal").textContent = "₹" + subtotal.toLocaleString("en-IN");
    document.getElementById("serviceFee").textContent = "₹" + fee.toLocaleString("en-IN");
    document.getElementById("gst").textContent = "₹" + gst.toLocaleString("en-IN");
    document.getElementById("total").textContent = "₹" + total.toLocaleString("en-IN");
    document.getElementById("priceBreakdown").classList.remove("d-none");
  }
  document.getElementById("checkIn")?.addEventListener("change", calcPrice);
  document.getElementById("checkOut")?.addEventListener("change", calcPrice);
</script>
